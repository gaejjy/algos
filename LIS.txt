void Swap(int* a , int * b)
{
	int temp;
	temp = *a;
	*a = *b;
	*b = temp;
}


int LIS(int Size, vector<int>& target,vector<int>& dpboard)
{
	int i,j,k,result;
	result = 0;
	for(i = 0;i<Size;i++)
	{
		for(j= 0;j<i;j++)
		{
			if(target[i]>target[j])
			{
				if(dpboard[i]<dpboard[j]+1)
				{
					dpboard[i] = dpboard[i]+1;
				}
			}
		}
		
		if(result < dpboard[i])
		{
			result = dpboard[i];
		}
	}
	return result; 
	
}
